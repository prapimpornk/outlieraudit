สูตรคำนวณ AdjRW — Thai DRG v6.3
⚠️ กรณี < 24 ชั่วโมง
# LOS น้อยกว่า 1,440 นาที
เงื่อนไข: LOS_นาที < 1440
AdjRW = RW0d
🟢 Low Outlier
# LOS < WtLOS/3 และ WtLOS > 3
เงื่อนไข: LOS < WtLOS/3
AdjRW = RW0d + LOS
*(RW - RW0d)
/ CEILING(WtLOS/3)
🔵 Inlier (ปกติ)
# WtLOS/3 ≤ LOS ≤ OT
เงื่อนไข: อยู่ในช่วงมาตรฐาน
AdjRW = RW ปกติ
# ตามตาราง b12/b23
🔴 High Outlier
# LOS > OT (OT = 3×WtLOS)
เงื่อนไข: LOS > OT
AdjRW = RW + OF
* b23 * (LOS - OT)
# OF, b23 จาก Appendix G
นิยามตัวแปร
LOS = วันนอนจริง หักวันลากลับบ้าน (LeaveDay × 24 ชม.)  |  WtLOS = วันนอนเฉลี่ยมาตรฐานของ DRG (Appendix G)
OT = 3 × WtLOS  |  RW0d = RW กรณี < 24 ชม.  |  OF = Outlier Factor เฉพาะ DRG  |  CEILING(x) = ปัดขึ้นเป็นจำนวนเต็ม
📌 สิ่งที่ต้อง Audit
🔴 High Outlier — ตรวจสอบ LOS จริง, วันลากลับบ้านที่ยังไม่หัก, เอกสาร Medical Necessity ครบ?
🟢 Low Outlier — มี CC/MCC ที่ยังไม่ได้บันทึก? จำหน่ายก่อนกำหนดจริงหรือไม่?
⚠️ < 24 ชม. — ยืนยัน admit จริง, ควรเป็น OPD หรือ IPD? สปสช. มักตรวจพิเศษกลุ่มนี้
HOSxP v3 — SQL Query สำหรับดึง Outlier
SELECT
    i.an                                             AS AN,
    i.hn                                             AS HN,
    CONCAT(p.pname, p.fname, ' ', p.lname)         AS ชื่อ_สกุล,
    DATE(i.regdate)                                  AS วันรับ,
    DATE(i.dchdate)                                  AS วันจำหน่าย,
    i.los                                            AS LOS_วัน,
    i.pdx                                            AS PDx,
    i.drg                                            AS DRG,
    i.rw                                             AS RW,
    i.adjrw                                          AS AdjRW,
    i.wtlos                                          AS WtLOS,
    ROUND(i.wtlos * 3, 2)                            AS OT,
    ROUND(i.wtlos / 3, 2)                            AS Low_Cut,
    CASE
        WHEN i.los < 1                          THEN 'น้อยกว่า 24 ชั่วโมง'
        WHEN i.wtlos > 3
             AND i.los < (i.wtlos / 3)         THEN 'Low Outlier'
        WHEN i.los > (i.wtlos * 3)              THEN 'High Outlier'
        ELSE                                         'Inlier'
    END                                              AS ประเภท_Outlier,
    ROUND(i.los - i.wtlos, 2)                        AS ส่วนต่างจาก_WtLOS,
    i.ward_name                                      AS Ward,
    i.dchtype                                        AS สถานะจำหน่าย,
    dc.name                                          AS แพทย์เจ้าของไข้,
    i.pttype                                         AS สิทธิ

FROM ipt i
LEFT JOIN patient p  ON p.hn = i.hn
LEFT JOIN doctor  dc ON dc.code = i.doctor

WHERE
    i.dchdate BETWEEN '2024-10-01 00:00:00'
              AND     '2025-09-30 23:59:59'

    AND i.pttype IN ('UC', 'UCS', 'A7', '89')

    -- เฉพาะเคส Outlier เท่านั้น
    AND (
        i.los < 1
        OR (i.wtlos > 3 AND i.los < (i.wtlos / 3))
        OR i.los > (i.wtlos * 3)
    )

    AND i.drg IS NOT NULL
    AND i.drg != ''

ORDER BY
    FIELD(
        CASE
            WHEN i.los < 1                   THEN 'zero'
            WHEN i.los > (i.wtlos * 3)      THEN 'high'
            WHEN i.los < (i.wtlos / 3)      THEN 'low'
            ELSE 'normal'
        END,
        'high', 'zero', 'low', 'normal'
    ), i.adjrw DESC;
📅 ปรับช่วงวันที่
ทั้งปีงบ 2568: '2024-10-01' → '2025-09-30'
ไตรมาส 1: '2024-10-01' → '2024-12-31'
เดือนเมษายน: '2025-04-01' → '2025-04-30'
🔧 ตัวกรองเพิ่มเติม
เฉพาะ High: i.los > (i.wtlos*3)
เฉพาะ Low: i.los < (i.wtlos/3)
เฉพาะ Ward: AND i.ward_name = 'MED'
